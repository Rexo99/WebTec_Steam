  "Serie": {
    "Titel": "Titel der Serie",
    "Genre": "Genre der Serie",
    "Staffeln": "Anzahl der Staffeln",
    "Platform": "Platform auf der die Serie zu finden ist",
    "Rating": "Rating der Serie",
    "Bemerkung": "Bemerkung zur Serie"
  }
}

suchbegriff //Body in Post-Anfrage (es wird keine Json mitgeschickt)
